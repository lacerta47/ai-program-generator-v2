```ts
import { NextRequest, NextResponse } from 'next/server';
import { adminAuth, adminDb } from '@/lib/firebase/admin';
import { todayKeyKST } from '@/lib/usageDay';
import { readEffectiveLimit } from '@/lib/admin/usageConfig';

export const runtime = 'nodejs';

export async function GET(req: NextRequest) {
  const authHeader = req.headers.get('authorization') ?? '';
  const idToken = authHeader.startsWith('Bearer ') ? authHeader.slice(7) : '';
  if (!idToken) return NextResponse.json({ error: '로그인이 필요해요.' }, { status: 401 });

  let uid: string;
  let isAdmin = false;
  try {
    const decoded = await adminAuth.verifyIdToken(idToken);
    uid = decoded.uid;
    isAdmin = decoded.admin === true;
  } catch {
    return NextResponse.json({ error: '로그인이 만료됐어요. 다시 로그인해 주세요.' }, { status: 401 });
  }

  if (isAdmin) {
    return NextResponse.json({ used: 0, limit: null, unlimited: true });
  }

  try {
    const day = todayKeyKST();
    const snap = await adminDb.collection('usage').doc(`${uid}_${day}`).get();
    const used = (snap.data()?.count as number | undefined) ?? 0;
    const limit = await readEffectiveLimit(uid);
    return NextResponse.json({ used, limit, unlimited: false });
  } catch (e) {
    console.error('내 사용량 조회 실패:', e);
    return NextResponse.json({ error: '사용량을 불러오지 못했어요.' }, { status: 500 });
  }
}
```

```tsx
/** 본인 토큰으로 /api/me/usage 호출. */
async function fetchMyUsage(): Promise<Usage> {
  const u = auth.currentUser;
  if (!u) throw new Error('로그인이 필요해요.');
  const idToken = await u.getIdToken();
  const res = await fetch('/api/me/usage', { headers: { Authorization: `Bearer ${idToken}` } });
  const data = await res.json().catch(() => ({}));
  if (!res.ok) throw new Error(data?.error || `요청 실패 (${res.status})`);
  return data as Usage;
}

function AccountCard({
  uid,
  email,
  createdAt,
}: {
  uid: string;
  email: string | null;
  createdAt?: string;
}) {
  const { toast } = useToast();
  const [nickname, setNickname] = useState<string | null>(null);
  const [usage, setUsage] = useState<Usage | null>(null);
  const [editOpen, setEditOpen] = useState(false);
  const [draft, setDraft] = useState('');
  const [busy, setBusy] = useState(false);

  useEffect(() => {
    getUserProfile(uid)
      .then((p) => setNickname(p?.nickname ?? null))
      .catch((e) => console.error('프로필 조회 실패:', e));
  }, [uid]);

  useEffect(() => {
    let alive = true;
    fetchMyUsage()
      .then((u) => {
        if (alive) setUsage(u);

      .catch((e) => console.error('사용량 조회 실패:', e));

        toast('그 별명은 쓸 수 없어요. 예쁜 말로 바꿔 볼까요?');

        toast("그 별명은 쓸 수 없어요. '관리자' 같은 말은 넣을 수 없어요.");

      } else {
        toast('별명을 바꾸지 못했어요. 잠시 후 다시 해주세요.');

      setBusy(false);

function MyWorks({ uid }: { uid: string }) {
  const { toast } = useToast();
  const [posts, setPosts] = useState<Post[] | null>(null);

    let alive = true;
    setError(false);

        if (alive) setPosts(p);
      })
      .catch((e) => {
        console.error('내 작품 불러오기 실패:', e);
        if (alive) setError(true);
      });
    return () => {
      alive = false;
    };
```

```javascript
// 마이페이지 데이터/사용량 통합 self-test — 실행 중인 dev 서버(localhost:3000) 대상.
// 생성물 자동 정리. 실행: node scripts/selftest-mypage.mjs
import { readFileSync } from 'node:fs';
import { initializeApp as initAdmin, cert } from 'firebase-admin/app';
import { getAuth as getAdminAuth } from 'firebase-admin/auth';
import { getFirestore as getAdminDb } from 'firebase-admin/firestore';
import { initializeApp } from 'firebase/app';
import { getAuth, signInWithCustomToken, signOut } from 'firebase/auth';

const env = Object.fromEntries(
  readFileSync(new URL('../.env.local', import.meta.url), 'utf8')
    .split('\n').filter((l) => l.includes('=')).map((l) => {
      const i = l.indexOf('='); return [l.slice(0, i).trim(), l.slice(i + 1).trim()];
    }),
);
const sa = JSON.parse(readFileSync(new URL('../serviceAccountKey.json', import.meta.url), 'utf8'));
initAdmin({ credential: cert(sa) });
const aAuth = getAdminAuth();
const aDb = getAdminDb();
const app = initializeApp({
  apiKey: env.NEXT_PUBLIC_FIREBASE_API_KEY,
  authDomain: env.NEXT_PUBLIC_FIREBASE_AUTH_DOMAIN,
  projectId: env.NEXT_PUBLIC_FIREBASE_PROJECT_ID,
});
const auth = getAuth(app);
const BASE = 'http://localhost:3000';
const S = Date.now();
const ME = `mypage-me-${S}`;
const OTHER = `mypage-other-${S}`;
const today = new Date(Date.now() + 9 * 3600 * 1000).toISOString().slice(0, 10);

let pass = 0, fail = 0;
const ok = (c, m) => { if (c) { pass++; console.log('  OK ', m); } else { fail++; console.log('  XX ', m); } };

const p1 = aDb.collection('posts').doc();
const p2 = aDb.collection('posts').doc();
const pOther = aDb.collection('posts').doc();
const usageId = `${ME}_${today}`;

try {
  // 내 글 2개(시간차) + 남의 글 1개
  await p1.set({ ownerUid: ME, title: '내작품1', authorName: 'me', categoryId: 'x', code: { html: '', css: '', javascript: '' }, prompt: '', createdAt: S - 1000 });
  await p2.set({ ownerUid: ME, title: '내작품2', authorName: 'me', categoryId: 'x', code: { html: '', css: '', javascript: '' }, prompt: '', createdAt: S });
  await pOther.set({ ownerUid: OTHER, title: '남작품', authorName: 'o', categoryId: 'x', code: { html: '', css: '', javascript: '' }, prompt: '', createdAt: S });

  // 1) ownerUid 쿼리: 내 글만, 최신순 (fetchMyPosts와 동일 shape — 배포된 인덱스 사용)
  const snap = await aDb.collection('posts').where('ownerUid', '==', ME).orderBy('createdAt', 'desc').limit(50).get();
  const ids = snap.docs.map((d) => d.id);
  ok(ids.length === 2 && !ids.includes(pOther.id), '내 글만 조회(남의 글 제외)');
  ok(ids[0] === p2.id, '최신순 정렬(내작품2 먼저)');

  // 2) /api/me/usage: 본인 사용량
  await aDb.doc(`usage/${usageId}`).set({ uid: ME, day: today, count: 4, updatedAt: Date.now() });
  await signInWithCustomToken(auth, await aAuth.createCustomToken(ME, {}));
  const meTok = await auth.currentUser.getIdToken();
  const r = await fetch(`${BASE}/api/me/usage`, { headers: { Authorization: `Bearer ${meTok}` } });
  const d = await r.json().catch(() => ({}));
  ok(r.status === 200 && d.used === 4 && typeof d.limit === 'number' && d.unlimited === false, `본인 used 4·limit 반환 (used=${d.used}, limit=${d.limit})`);

  // 3) 토큰 없음 → 401
  const r401 = await fetch(`${BASE}/api/me/usage`);
  ok(r401.status === 401, `토큰없음 401 (got ${r401.status})`);

  // 4) admin → 무제한
  await signOut(auth);
  await signInWithCustomToken(auth, await aAuth.createCustomToken('mypage-admin', { admin: true }));
  const adminTok = await auth.currentUser.getIdToken();
  const ra = await fetch(`${BASE}/api/me/usage`, { headers: { Authorization: `Bearer ${adminTok}` } });
  const da = await ra.json().catch(() => ({}));
  ok(ra.status === 200 && da.unlimited === true, 'admin → 무제한');
} catch (e) {
  fail++; console.error('스크립트 예외:', e);
} finally {
  for (const ref of [p1, p2, pOther]) { try { await ref.delete(); } catch {} }
  try { await aDb.doc(`usage/${usageId}`).delete(); } catch {}
  await signOut(auth).catch(() => {});
  console.log(`\n결과: ${pass} pass / ${fail} fail`);
  process.exit(fail ? 1 : 0);
}
```
